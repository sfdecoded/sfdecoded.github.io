    public static List<Account> getActiveAccounts() {
        return [
            SELECT Id, Name, Industry, AnnualRevenue
            FROM Account
            WHERE IsActive__c = true
            ORDER BY AnnualRevenue DESC
            LIMIT 200
        ];
    }
}
